create table device_property(
	id int(11) not null primary key auto_increment comment '自增长主键',
	device varchar(20) not null default '' comment '设备',
	app_id varchar(20) not null default '0' comment '应用id',
	app_name varchar(20) not null default '' comment '应用名称',
	platform int(3) not null default 0 comment '平台：0：未知，1：ios,2：android',
	device_UUID varchar(50) not null default '0' comment '设备唯一id'
)
COMMENT='APP设备基本属性'
COLLATE='utf8_general_ci'
ENGINE=InnoDB;

create index index_appp_id on device_property(app_id);
create index index_device_UUID on device_property(device_UUID);

create table performance(
	id int(11) not null primary key auto_increment comment '自增长主键',
	fps double(10,2) default 0.00 comment '帧数',
	cpu double(10,2) default 0.00 comment 'cpu使用率',
	memory double(10,2) default 0.00 comment 'memory使用率',
	warnning_fps_page varchar(20) not null default '' comment 'fps警告',
	warning_memory_page varchar(20) not null default '' comment 'memory警告',
	warning_cpu_page varchar(20) not null default '' comment 'cpu警告',
	record_time datetime not null comment '记录时间',
	device_UUID varchar(50) not null default '0' comment '设备唯一id',
	device varchar(50) not null default '0' comment '设备',
	app_id varchar(50) not null default '0' comment '应用id',
	platform int(3) default 0 comment '平台'
)
COMMENT='APP性能数据'
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
create index index_app_id on performance(app_id);
create index index_record_time on performance(record_time);
create index index_device_UUID on performance(device_UUID);

create table start_app(
	id int(11) not null primary key auto_increment comment '自增长主键',
	start_time datetime not null comment '启动时间',
	duriation_time varchar(50) default '' comment '启动时间段',
	device_UUID varchar(50) not null default '0' comment '设备唯一id',
	device varchar(50) not null default '0' comment '设备',
	app_id varchar(50) not null default '0' comment '应用id',
	platform int(3) default 0 comment '平台'
)
COMMENT='APP启动数据'
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
create index index_app_id on start_app(app_id);
create index index_start_time on start_app(start_time);
create index index_device_UUID on start_app(device_UUID);

create table network_log(
	id int(11) not null primary key auto_increment comment '自增长主键',
	uuid varchar(50) not null default '' comment 'uuid',
	start_date datetime not null comment '记录开始时间',
	end_date datetime comment '记录结束时间',
	req_url varchar(50) default '' comment '请求url',
	relative_url varchar(50) default '' comment '相对url',
	req_timeout_interval Double(10,2) default 0.00 comment '超时间隔',
	req__http_header varchar(50) default '' comment '请求http头',
	req_body varchar(50) default '' comment '请求体',
	resp_type varchar(50) default '' comment '反馈类型',
	resp_content_length int(11) default 0 comment '反馈内容长度',
	resp_encode_name varchar(50) default '' comment '反馈编码',
	resp_file_name varchar(50) default '' comment '反馈文件',
	resp_status_code varchar(50) default '' comment '反馈状态码',
	resp_header varchar(50) default '' comment '反馈头',
	device_UUID varchar(50) not null default '0' comment '设备唯一id',
	device varchar(50) not null default '0' comment '设备',
	app_id varchar(50) not null default '0' comment '应用id',
	platform int(3) default 0 comment '平台'
)
COMMENT='APP日志数据'
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
create index index_app_id on network_log(app_id);
create index index_start_date on network_log(start_date);
create index index_device_UUID on network_log(device_UUID);

create table crash(
	id int(11) not null primary key auto_increment comment '自增长主键',
	crash_queue varchar(50) default '' comment '崩溃队列',
	crash_time datetime not null comment '崩溃时间',
	report_time datetime comment '上报时间',
	bundle_name varchar(20) default '' comment 'bundle名	',
	bundle_version varchar(50) default '' comment '版本',
	use_time datetime comment '使用时间',
	device_type varchar(50) default '' comment '设备类型',
	crash_desc varchar(50) default '' comment '崩溃描述',
	crash_thread varchar(50) default '' comment '崩溃线程',
	device_UUID varchar(50) not null default '0' comment '设备唯一id',
	device varchar(50) not null default '0' comment '设备',
	app_id varchar(50) not null default '0' comment '应用id',
	platform int(3) default 0 comment '平台'
)
COMMENT='APP崩溃数据'
COLLATE='utf8_general_ci'
ENGINE=InnoDB;
create index index_app_id on crash(crash_time);
create index index_crash_time on crash(crash_time);
create index index_device_UUID on crash(device_UUID);
